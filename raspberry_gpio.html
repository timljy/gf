LEDを点灯させるプログラム python用


import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM) #GPIOへアクセスする番号をBCMの番号で指定することを宣言します。
GPIO.setup(15,GPIO.OUT) #BCMの15番ピン、物理的には10番ピンを出力に設定します。
GPIO.setup(2,GPIO.IN) #BCM 2番ピンを入力に設定します。
try:
while True:
if GPIO.input(2) == GPIO.LOW:
GPIO.output(15,GPIO.HIGH)
else:
GPIO.output(15,GPIO.LOW)
time.sleep(0.1)
except KeyboardInterrupt:
GPIO.cleanup()

pinMode(0, OUTPUT);
pinMode(1, INPUT);
// Turn pin 0 on then off
digitalWrite(0, HIGH);
digitalWrite(0, LOW);
// Read input from pin 1
int status = digitalRead(1);
printf("Pin Input = %d\n", status);

}
